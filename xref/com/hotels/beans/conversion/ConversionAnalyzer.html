17  package com.hotels.beans.conversion;
18  
19  import static java.util.Optional.empty;
20  import static java.util.Optional.of;
21  
22  import static com.hotels.beans.cache.CacheManagerFactory.getCacheManager;
23  import static com.hotels.beans.conversion.ConversionProcessorFactory.getConversionProcessor;
24  
25  import java.util.Optional;
26  import java.util.function.Function;
27  
28  import com.hotels.beans.cache.CacheManager;
29  import com.hotels.beans.utils.ClassUtils;
30  
31  /**
32   * This class analyzes the two classes received in input, and returns the conversion processor to be used.
33   */
34  public final class ConversionAnalyzer {
35      /**
36       * CacheManager instance {@link CacheManager}.
37       */
38      private static final CacheManager CACHE_MANAGER = getCacheManager("conversionAnalyzer");
39  
40      /**
41       * Class utils instance {@link ClassUtils}.
42       */
43      private final ClassUtils classUtils;
44  
45      /**
46       * Default constructor.
47       */
48      public ConversionAnalyzer() {
49          this.classUtils = new ClassUtils();
50      }
51  
52      /**
53       * Analyzes Fields given as input and returns the conversion processor.
54       * @param sourceFieldType source field class
55       * @param destinationFieldType the destination field class
56       * @return an {@link Optional} containing the conversion function (if exists)
57       */
58      @SuppressWarnings("unchecked")
59      public Optional<Function<?, ?>> getConversionFunction(final Class<?> sourceFieldType, final Class<?> destinationFieldType) {
60          final String cacheKey = "ConversionFunction-" + sourceFieldType.getName() + "-" + destinationFieldType.getName();
61          return CACHE_MANAGER.getFromCache(cacheKey, Optional.class).orElseGet(() -> {
62              Optional<Function<?, ?>> conversionFunction = empty();
63              if (destinationFieldType != sourceFieldType && classUtils.isPrimitiveType(sourceFieldType) && classUtils.isPrimitiveType(destinationFieldType)) {
64                  conversionFunction = getConversionFunction(getConversionProcessor(destinationFieldType), sourceFieldType);
65              }
66              CACHE_MANAGER.cacheObject(cacheKey, conversionFunction);
67              return conversionFunction;
68          });
69      }
70  
71      /**
72       * Returns the type transformer function for the given type.
73       * @param conversionProcessor the {@link ConversionProcessor} for the given type
74       * @param sourceFieldType the source field class
75       * @return the conversion function
76       */
77      @SuppressWarnings("unchecked")
78      private Optional<Function<?, ?>> getConversionFunction(final ConversionProcessor conversionProcessor, final Class<?> sourceFieldType) {
79          final String cacheKey = "ConversionFunction-" + sourceFieldType.getName();
80          return CACHE_MANAGER.getFromCache(cacheKey, Optional.class).orElseGet(() -> {
81              Optional<Function<?, ?>> conversionFunction;
82              if (sourceFieldType == String.class) {
83                  conversionFunction = of(conversionProcessor.convertString());
84              } else if (sourceFieldType == Byte.class) {
85                  conversionFunction = of(conversionProcessor.convertByte());
86              } else if (sourceFieldType == byte.class) {
87                  conversionFunction = of(conversionProcessor.convertPrimitiveByte());
88              } else if (sourceFieldType == Short.class) {
89                  conversionFunction = of(conversionProcessor.convertShort());
90              } else if (sourceFieldType == short.class) {
91                  conversionFunction = of(conversionProcessor.convertPrimitiveShort());
92              } else if (sourceFieldType == Integer.class) {
93                  conversionFunction = of(conversionProcessor.convertInteger());
94              } else if (sourceFieldType == int.class) {
95                  conversionFunction = of(conversionProcessor.convertInt());
96              } else if (sourceFieldType == Long.class) {
97                  conversionFunction = of(conversionProcessor.convertLong());
98              } else if (sourceFieldType == long.class) {
99                  conversionFunction = of(conversionProcessor.convertPrimitiveLong());
100             } else if (sourceFieldType == Float.class) {
101                 conversionFunction = of(conversionProcessor.convertFloat());
102             } else if (sourceFieldType == float.class) {
103                 conversionFunction = of(conversionProcessor.convertPrimitiveFloat());
104             } else if (sourceFieldType == Double.class) {
105                 conversionFunction = of(conversionProcessor.convertDouble());
106             } else if (sourceFieldType == double.class) {
107                 conversionFunction = of(conversionProcessor.convertPrimitiveDouble());
108             } else if (sourceFieldType == Character.class) {
109                 conversionFunction = of(conversionProcessor.convertCharacter());
110             } else if (sourceFieldType == char.class) {
111                 conversionFunction = of(conversionProcessor.convertChar());
112             } else if (sourceFieldType == Boolean.class) {
113                 conversionFunction = of(conversionProcessor.convertBoolean());
114             } else if (sourceFieldType == boolean.class) {
115                 conversionFunction = of(conversionProcessor.convertPrimitiveBoolean());
116             } else {
117                 conversionFunction = empty();
118             }
119             CACHE_MANAGER.cacheObject(cacheKey, conversionFunction);
120             return conversionFunction;
121         });
122     }
123 }
