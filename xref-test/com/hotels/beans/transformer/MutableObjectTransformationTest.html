41  /**
42   * Unit test for all {@link Transformer} functions related to Mutable type Java Beans.
43   */
44  public class MutableObjectTransformationTest extends AbstractTransformerTest {
45      /**
46       * The class to be tested.
47       */
48      @InjectMocks
49      private TransformerImpl underTest;
50  
51      /**
52       * Initialized mocks.
53       */
54      @BeforeMethod
55      public void beforeMethod() {
56          initMocks(this);
57      }
58  
59      /**
60       * Test that an exception is thrown if there is no default constructor defined for the mutable bean object.
61       */
62      @Test(expectedExceptions = InvalidBeanException.class)
63      public void testTransformThrowsExceptionWhenMutableBeanHasNoDefaultConstructor() {
64          //GIVEN
65  
66          //WHEN
67          underTest.transform(fromFoo, MutableToFooInvalid.class);
68      }
69  
70      /**
71       * Test mutable beans are correctly copied.
72       */
73      @Test
74      public void testMutableBeanIsCorrectlyCopied() {
75          //GIVEN
76  
77          //WHEN
78          MutableToFoo actual = underTest.transform(fromFoo, MutableToFoo.class);
79  
80          //THEN
81          assertThat(actual, sameBeanAs(fromFoo));
82      }
83  
84      /**
85       * Test transformation on an existing bean is correctly copied.
86       */
87      @Test
88      public void testTransformationOnAnExistingDestinationWorksProperly() {
89          //GIVEN
90          MutableToFooSubClass mutableToFoo = new MutableToFooSubClass();
91  
92          //WHEN
93          underTest.transform(fromFooSubClass, mutableToFoo);
94  
95          //THEN
96          assertThat(mutableToFoo, sameBeanAs(fromFooSubClass));
97      }
98  
99      /**
100      * Test that a given field transformer function is applied only to a specific field even if there are other ones with same name.
101      */
102     @Test
103     public void testFieldTransformationIsAppliedOnlyToASpecificField() {
104         //GIVEN
105         String namePrefix = "prefix-";
106         FieldTransformer<String, String> nameTransformer = new FieldTransformer<>("nestedObject.name", val -> namePrefix + val);
107 
108         //WHEN
109         MutableToFoo actual = underTest
110                 .withFieldTransformer(nameTransformer)
111                 .transform(fromFoo, MutableToFoo.class);
112 
113         //THEN
114         assertEquals(fromFoo.getName(), actual.getName());
115         assertEquals(namePrefix + fromFoo.getNestedObject().getName(), actual.getNestedObject().getName());
116         underTest.resetFieldsTransformer();
117     }
118 
119     /**
120      * Test that a given field transformer function is applied to all field matching the given name when {@code flatFieldNameTransformation} is set to true.
121      */
122     @Test
123     public void testFieldTransformationIsAppliedToAllMatchingFields() {
124         //GIVEN
125         String namePrefix = "prefix-";
126         FieldTransformer<String, String> nameTransformer = new FieldTransformer<>(NAME_FIELD_NAME, val -> namePrefix + val);
127 
128         //WHEN
129         MutableToFoo actual = underTest
130                 .setFlatFieldNameTransformation(true)
131                 .withFieldTransformer(nameTransformer)
132                 .transform(fromFoo, MutableToFoo.class);
133 
134         //THEN
135         assertEquals(namePrefix + fromFoo.getName(), actual.getName());
136         assertEquals(namePrefix + fromFoo.getNestedObject().getName(), actual.getNestedObject().getName());
137         underTest.resetFieldsTransformer();
138     }
139 
140     /**
141      * Test that the transformer is able to copy object even if the source object has no setter methods and the destination
142      * object has no setter methods.
143      */
144     @Test
145     public void testTransformerIsAbleToCopyObjectsWithoutRequiredMethods() {
146         //GIVEN
147         FromFooSimpleNoGetters fromFooSimpleNoGetters = new FromFooSimpleNoGetters(NAME, ID);
148 
149         //WHEN
150         MutableToFooSimpleNoSetters actual = underTest.transform(fromFooSimpleNoGetters, MutableToFooSimpleNoSetters.class);
151 
152         //THEN
153         assertThat(actual, sameBeanAs(fromFooSimpleNoGetters));
154     }
155 
156     /**
157      * Test that a bean containing a field not existing in the source object, but with a transformer function defined for such object is correctly copied.
158      */
159     @Test
160     public void testThatAnyTypeOfBeanContainsANotExistingFieldInTheSourceObjectIsCorrectlyCopiedThroughTransformerFunctions() {
161         //GIVEN
162         FromFooSimple fromFooSimple = new FromFooSimple(NAME, ID);
163         FieldTransformer<Object, Integer> ageFieldTransformer = new FieldTransformer<>(AGE_FIELD_NAME, val -> AGE);
164 
165         //WHEN
166         underTest.withFieldTransformer(ageFieldTransformer);
167         MutableToFooNotExistingFields mutableObjectBean = underTest.transform(fromFooSimple, MutableToFooNotExistingFields.class);
168 
169         //THEN
170         assertThat(mutableObjectBean, hasProperty(AGE_FIELD_NAME, equalTo(AGE)));
171     }
